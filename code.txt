<script>
    function selectChoice(selected) {
        var choices = document.querySelectorAll('.choice');
        choices.forEach(function(choice) {
            if (choice.textContent === selected) {
                choice.classList.add('selected');
            } else {
                choice.classList.remove('selected');
            }
        });
    }
</script>
